import express from "express"; import fetch from "node-fetch"; const app = express(); app.use(express.json()); app.post("/chat", async (req, res) => { const userMsg = req.body.message; const r = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${process.env.OPENAI_API_KEY}` }, body: JSON.stringify({ model: "gpt-4o-mini", messages: [{ role: "user", content: userMsg }] }) }); const data = await r.json(); res.json({ reply: data.choices[0].message.content }); }); app.listen(3000);
